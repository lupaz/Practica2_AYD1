    public String Suma(String a, String b) {
        String resultado;
        try{
            int a1 = Integer.parseInt(a, 2);
            int b1 = Integer.parseInt(b, 2);
            int c1 = a1 + b1;
            resultado = Integer.toString(c1, 2);
        } 
        catch(Exception e){
            resultado = "Error";
        }  
        return resultado;
    }
